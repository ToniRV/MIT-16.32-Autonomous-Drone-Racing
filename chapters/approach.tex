\textbf{Constraints:} 
\begin{itemize}
  \item Input constraints:

  \item Path constraints:
  $\forall t, t_{0} \leq t \leq t_{f}$ 
  $$c(x(t), u(t), t) \leq 0,$$
\end{itemize}
